// Step 1: Include the standard input-output library
#include <stdio.h>

// Step 2: Define the main function
int main() {
    // Step 3: Declare variables to store user input
    char name[50];   // String for name (array of characters)
    int age;         // Integer for age
    float salary;    // Float for salary

    // Step 4: Ask the user for their name
    printf("Enter your name: ");
    scanf("%s", name);   // %s is used for strings

    // Step 5: Ask the user for their age
    printf("Enter your age: ");
    scanf("%d", &age);   // %d is used for integers

    // Step 6: Ask the user for their salary
    printf("Enter your salary: ");
    scanf("%f", &salary);   // %f is used for floating-point numbers

    // Step 7: Print the collected information neatly
    printf("\n--- User Information ---\n");
    printf("Name: %s\n", name);
    printf("Age: %d\n", age);
    printf("Salary: %.2f\n", salary); // %.2f prints salary with 2 decimal places

    // Step 8: End program successfully
    return 0;
}
