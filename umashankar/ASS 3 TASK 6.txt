#include <stdio.h>

int main() {
    int n;
    
    // Read number of elements
    printf("Enter number of elements: ");
    scanf("%d", &n);

    int arr[n];   // No hardcoded size (VLA)

    // Read elements into array
    printf("Enter %d elements:\n", n);
    for (int i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }

    int sum = 0;
    int largest = arr[0];
    int smallest = arr[0];

    // Calculate sum, largest, and smallest
    for (int i = 0; i < n; i++) {
        sum += arr[i];

        if (arr[i] > largest)
            largest = arr[i];

        if (arr[i] < smallest)
            smallest = arr[i];
    }

    float average = (float)sum / n;

    // Output results
    printf("Sum = %d\n", sum);
    printf("Average = %.2f\n", average)
