#include <stdio.h>

int main() {
    FILE *fp;
    char ch;

    /* Writing text to a file */
    fp = fopen("data.txt", "w");
    if (fp == NULL) {
        printf("Error opening file for writing.\n");
        return 1;
    }
    fprintf(fp, "This is the first line in the file.\n");
    fclose(fp);

    /* Reading text from the file */
    fp = fopen("data.txt", "r");
    if (fp == NULL) {
        printf("Error opening file for reading.\n");
        return 1;
    }
    printf("File contents after writing:\n");
    while ((ch = fgetc(fp)) != EOF) {
        printf("%c", ch);
    }
    fclose(fp);

    /* Appending text to the file */
    fp = fopen("data.txt", "a");
    if (fp == NULL) {
        printf("Error opening file for appending.\n");
        return 1;
    }
    fprintf(fp, "This line is appended to the file.\n");
    fclose(fp);

    /* Reading file again after appending */
    fp = fopen("data.txt", "r");
    if (fp == NULL) {
        printf("Error opening file.\n");
        return 1;
    }
    printf("\nFile contents after appending:\n");
    while ((ch = fgetc(fp)) != EOF) {
        printf("%c", ch);
    }
    fclose(fp);

    return 0;
}
